iota0 x = iota x - 1
fact n = * iota n
ave x = + x / length x
# variance (分散)
var x = ave (x ^ 2) - ave x ^ 2
# standard deviation (標準偏差)
sd x = sqrt var x
# T-score (偏差値)
tscore x = (x - ave x) / sd x * 10 + 50
